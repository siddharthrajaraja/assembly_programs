CREATE TABLE PATIENT (
PID NUMBER (4) PRIMARY KEY,
PNAME VARCHAR (20) NOT NULL,
PAGE NUMBER(3) NOT NULL,
SEX CHAR(1) NOT NULL,
CONTACTNO NUMBER UNIQUE,
ADDR VARCHAR(40) DEFAULT 'mumbai',
CHECK (PAGE>0),
CHECK (CONTACTNO=10));
